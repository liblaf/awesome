export const Repos = ({ group }) => {
  const { repos } = useData();
  return (
    <table>
      <thead>
        <tr>
          <th> Name </th>
          <th> Stars </th>
          <th> Commits </th>
          <th> Description </th>
        </tr>
      </thead>
      <tbody>
        {repos[group].map((repo) => (
          <tr>
            <td>
              <a href={repo.url}> {repo.name} </a>
            </td>
            <td align="right"> {formatNumber(repo.stars)} </td>
            <td align="right"> {formatNumber(repo.commits)} </td>
            <td> {repo.description} </td>
          </tr>
        ))}
      </tbody>
    </table>
  );
};
